      https.get('https://api.koulusafka.fi/info/schools.php', (resp) => {
        var data = '';
        var koulut = '';

        resp.on('data', (chunk) => {
          data += chunk;
        });

        resp.on('end', () => {
          jsondata = JSON.parse(data);

          koulut = 'Helsingin koulut:\n';
          for (var i = 0; i < jsondata.regions[0].values.length; i++) {
            koulut += "<tr>";
            koulut += "   <td>" + jsondata.regions[0].values[i].name + "</td>\n   <td>" + jsondata.regions[0].values[i].id + "</td>\n";
            koulut += "</tr>";
          }
          console.log(koulut);
        });
    });
